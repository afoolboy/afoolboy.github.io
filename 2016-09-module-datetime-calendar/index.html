我们要统计的日期具有这样的性质，它明年的今天，是星期五。
所以，我们的思路是读取文件，格式化为datetime对象，判断一年后的今天是否为星期五。

#coding=utf-8

import locale
import calendar
from datetime import datetime


"""
Friday is the best day of the week, and so I really want to know how many Fridays there are in this file. But, with a twist. I want to know how many Fridays there are one year later than each date."""

locale.setlocale(locale.LC_ALL,'en_US.UTF-8')
calendar.setfirstweekday(calendar.MONDAY)

date_format = "%B %d, %Y"

count = 0

with open("date.txt") as date_file:
    for line in date_file:
        date_str = line.strip()
        date_obj = datetime.strptime(date_str,date_format)

        if not (calendar.isleap(date_obj.year) and date_obj.day == 29):
            date_obj = date_obj.replace(year = date_obj.year +1)

            if date_obj.isoweekday() == 5:
                count += 1
                pass
        elif int(date_obj.strftime("%m")) == 2:
            print "find one 2,29"

print "abctf{%d}"%(count,)
